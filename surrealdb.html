⚡Instalasi

Linux / macOS (curl)

# Install terbaru
curl --proto '=https' --tlsv1.2 -sSf https://install.surrealdb.com | sh

# Install versi spesifik
curl -sSf https://install.surrealdb.com | sh -s -- --nightly

Docker

# In-memory (dev)
docker run --rm --pull always --name surrealdb \
  -p 8000:8000 surrealdb/surrealdb:1.4 \
  start --log debug --user root --pass root memory

# Persistent (file)
docker run -d --name surrealdb \
  -p 8000:8000 \
  -v $(pwd)/data:/data \
  surrealdb/surrealdb:1.4 \
  start --user root --pass root file:/data/db

Windows (PowerShell)

iwr https://windows.surrealdb.com -useb | iex

Homebrew (macOS)

brew install surrealdb/tap/surreal

Menjalankan Server

# Lokal dev — in-memory
surreal start --log debug --user root --pass root memory

# Produksi — RocksDB on disk
surreal start --bind 0.0.0.0:8000 \
              --user root --pass root \
              rocksdb:/path/to/database

# Cluster TiKV
surreal start tikv://tikv-host:2379

🚀Quick Start

-- 1. Pilih namespace dan database
USE NS myapp DB production;

-- 2. Buat tabel dan record
CREATE person:tobie SET
    name = 'Tobie',
    age  = 30,
    tags = ['developer', 'rust'];

-- 3. Query
SELECT * FROM person WHERE age > 18;

-- 4. Graph relation
RELATE person:tobie -> knows -> person:jaime
    SET since = time::now();

-- 5. Graph traversal
SELECT ->knows->person.name AS friends
FROM person:tobie;

✨Apa yang Baru di v1.4

🆕 Fitur Baru v1.4.0

Rilis April 2024 — membawa tipe tabel terstruktur, klausa IF NOT EXISTS, dukungan insert massal di Rust SDK, dan berbagai perbaikan bug.

1. Tipe Tabel Khusus (TYPE clause)

Sekarang dapat mendefinisikan tabel sebagai TYPE RELATION untuk edge graph, TYPE NORMAL untuk record biasa, atau TYPE ANY (default).

-- Tabel edge dengan TYPE RELATION
DEFINE TABLE assigned_to SCHEMAFULL
    TYPE RELATION
    IN tag
    OUT sticky
    PERMISSIONS
        FOR create, select, update, delete
            WHERE in.owner == $auth.id;

-- Tabel record biasa
DEFINE TABLE product SCHEMAFULL TYPE NORMAL;

-- Perilaku lama (default)
DEFINE TABLE legacy TYPE ANY;

2. IF NOT EXISTS

Semua pernyataan DEFINE kini mendukung IF NOT EXISTS — tidak error jika resource sudah ada.

DEFINE TABLE IF NOT EXISTS person SCHEMALESS;
DEFINE FIELD IF NOT EXISTS email ON person TYPE string;
DEFINE INDEX IF NOT EXISTS idx_email ON person FIELDS email UNIQUE;
DEFINE NAMESPACE IF NOT EXISTS myapp;
DEFINE DATABASE IF NOT EXISTS production;

3. Bulk Insert di Rust SDK

// Insert banyak record sekaligus
db.insert(Resource::from("person"))
    .content(vec![
        User { id: thing("person:tobie")?, name: "Tobie" },
        User { id: thing("person:jaime")?, name: "Jaime" },
    ])
    .await?;

🔍SELECT

Pernyataan SELECT digunakan untuk mengambil record dari satu atau beberapa tabel.

Sintaks Lengkap

SELECT [ VALUE ] @projections
FROM @targets
[ WITH [ NOINDEX | INDEX @indexes ] ]
[ WHERE @condition ]
[ SPLIT [ AT ] @field ]
[ GROUP [ BY ] @fields ]
[ ORDER [ BY ] @fields [ ASC | DESC ] [ COLLATE ] [ NUMERIC ] ]
[ LIMIT [ BY ] @limit ]
[ START [ AT ] @start ]
[ FETCH @fields ]
[ TIMEOUT @duration ]
[ PARALLEL ]
[ EXPLAIN [ FULL ] ]

Contoh SELECT

-- Semua field
SELECT * FROM person;

-- Field tertentu + alias
SELECT name, age, email AS contact FROM person;

-- Nested field
SELECT name.first, name.last FROM person;

-- WHERE dengan kondisi
SELECT * FROM person
WHERE age >= 18 AND active = true;

-- LIMIT + START (pagination)
SELECT * FROM person ORDER BY name LIMIT 10 START 20;

-- GROUP BY dengan agregasi
SELECT country, count() AS total
FROM person GROUP BY country;

-- VALUE: kembalikan array nilai tunggal
SELECT VALUE name FROM person;
-- Output: ["Alice", "Bob", ...]

-- Record tunggal (ONLY)
SELECT * FROM ONLY person:alice;

-- FETCH untuk resolve record link
SELECT *, author FROM article FETCH author;

-- Graph traversal
SELECT ->knows->person AS friends FROM person:tobie;
SELECT <-wrote<-person AS authors FROM article;

-- Multi-depth traversal
SELECT ->knows->person->knows->person AS friends_of_friends
FROM person:tobie;

-- SPLIT untuk flatten array
SELECT * FROM person SPLIT tags;

-- EXPLAIN query plan
EXPLAIN SELECT * FROM person WHERE email = 'test@email.com';

-- Subquery
SELECT *,
    (SELECT count() FROM order WHERE customer = $parent.id) AS order_count
FROM customer;

-- Range scan (efisien untuk ID terurut)
SELECT * FROM log:2024010100..2024013123;

-- Beberapa tabel sekaligus
SELECT * FROM user, person, admin;

-- TIMEOUT
SELECT * FROM bigdata TIMEOUT 5s;

➕CREATE

Sintaks

CREATE [ ONLY ] @targets
[ CONTENT @value | SET @fields | UNSET @fields ]
[ RETURN [ NONE | BEFORE | AFTER | DIFF | @projections ] ]
[ TIMEOUT @duration ]
[ PARALLEL ]
-- ID auto-generated (UUID/ULID)
CREATE person;

-- ID eksplisit + SET
CREATE person:alice SET
    name = 'Alice',
    age  = 28,
    tags = ['admin'];

-- CONTENT (dari objek JSON)
CREATE person CONTENT {
    name: 'Bob',
    age:  32,
    email: 'bob@example.com'
};

-- RETURN NONE (lebih cepat, tidak kembalikan data)
CREATE log SET msg = 'hello' RETURN NONE;

-- RETURN BEFORE / AFTER / DIFF
CREATE person:test SET score = 0 RETURN AFTER;

-- ID khusus (string, int, objek)
CREATE product:`samsung-s24`;
CREATE event:⟨2024-01-15T00:00:00Z⟩;
CREATE config:{env: "prod", region: "asia"};

📥INSERT

Sintaks

INSERT [ IGNORE ] INTO @table
    [ @values | (SELECT ...) ]
[ ON DUPLICATE KEY UPDATE @fields ]
[ RETURN ... ]
[ TIMEOUT @duration ]
-- Insert satu record
INSERT INTO person { name: 'Alice', age: 28 };

-- Insert banyak sekaligus (v1.4 ✓)
INSERT INTO person [
    { name: 'Alice', age: 28 },
    { name: 'Bob',   age: 32 },
    { name: 'Carol', age: 25 }
];

-- ON DUPLICATE KEY UPDATE (upsert by ID)
INSERT INTO person { id: person:alice, age: 29 }
ON DUPLICATE KEY UPDATE age = age + 1;

-- IGNORE: skip jika sudah ada
INSERT IGNORE INTO user { id: user:1, name: 'Admin' };

-- Insert dari SELECT
INSERT INTO archive (SELECT * FROM log WHERE age > 30d);

✏️UPDATE & UPSERT

Sintaks UPDATE

UPDATE [ ONLY ] @targets
[ CONTENT @value | MERGE @value | PATCH @value | SET @fields | UNSET @fields ]
[ WHERE @condition ]
[ RETURN ... ]
[ TIMEOUT @duration ]
-- Update satu record
UPDATE person:alice SET age = 29;

-- Update banyak record dengan kondisi
UPDATE person SET active = false
WHERE last_login < time::now() - 90d;

-- MERGE: gabung dengan data lama
UPDATE person:alice MERGE {
    address: { city: 'Jakarta' }
};

-- PATCH (JSON Patch RFC 6902)
UPDATE person:alice PATCH [
    { op: 'replace', path: '/age', value: 30 }
];

-- Increment / decrement
UPDATE product:123 SET stock -= 1, views += 1;

-- UNSET (hapus field)
UPDATE person UNSET phone;

-- Tipe casting saat update
UPDATE person SET
    waist  = <int>    "34",
    height = <float>  201,
    score  = <decimal> 0.3 + 0.3 + 0.3 + 0.1;

UPSERT

Sama seperti UPDATE tapi membuat record baru jika tidak ada.

UPSERT person:alice SET
    name   = 'Alice',
    visits += 1;

🗑️DELETE

Sintaks

DELETE [ ONLY ] @targets
[ WHERE @condition ]
[ RETURN ... ]
[ TIMEOUT @duration ]
-- Hapus satu record
DELETE person:alice;

-- Hapus dengan kondisi
DELETE log WHERE created_at < time::now() - 30d;

-- Hapus semua record di tabel
DELETE session;

-- RETURN BEFORE (kembalikan data sebelum dihapus)
DELETE person:bob RETURN BEFORE;

🕸️RELATE (Graph)

Membuat edge antara dua record (node). Edge sendiri adalah record dengan field in dan out.

Sintaks

RELATE [ ONLY ] @from -> @table -> @to
[ CONTENT @value | SET @fields ]
[ RETURN ... ]
[ TIMEOUT @duration ]
-- Hubungkan dua person
RELATE person:alice ->knows-> person:bob
    SET since = time::now();

-- Dengan data tambahan
RELATE person:alice ->purchased-> product:123
    CONTENT {
        quantity: 2,
        price:    99.99,
        at:       time::now()
    };

-- Query forward traversal
SELECT ->purchased->product.* AS items
FROM person:alice;

-- Query backward traversal
SELECT <-purchased<-person.name AS buyers
FROM product:123;

-- Query data pada edge
SELECT ->purchased[WHERE quantity > 1]->product.*
FROM person:alice;

-- Multi-hop
SELECT ->knows->person->purchased->product
FROM person:alice;

📋DEFINE TABLE

Sintaks Lengkap

DEFINE TABLE [ IF NOT EXISTS ] @name
    [ DROP ]
    [ SCHEMAFULL | SCHEMALESS ]
    [ TYPE [ NORMAL | RELATION [ IN @tables ] [ OUT @tables ] | ANY ] ]
    [ AS SELECT ... FROM ... ]  -- computed/view table
    [ CHANGEFEED @duration ]
    [ PERMISSIONS @permissions ]
    [ COMMENT @string ]
-- Schemaless (default)
DEFINE TABLE person SCHEMALESS;

-- Schemafull
DEFINE TABLE product SCHEMAFULL;

-- TYPE RELATION (v1.4) — edge graph
DEFINE TABLE likes SCHEMAFULL TYPE RELATION
    IN person
    OUT post | comment;  -- boleh beberapa tabel

-- TYPE NORMAL (v1.4)
DEFINE TABLE article SCHEMAFULL TYPE NORMAL;

-- DROP: data tidak disimpan, hanya agregasi
DEFINE TABLE reading DROP;

-- Computed/view table (pre-aggregated)
DEFINE TABLE temp_by_city AS
    SELECT
        city,
        math::mean(temperature) AS avg_temp,
        count() AS readings
    FROM sensor
    GROUP BY city;

-- CHANGEFEED (untuk real-time CDC)
DEFINE TABLE order CHANGEFEED 1d;

-- Permissions
DEFINE TABLE post SCHEMAFULL
    PERMISSIONS
        FOR select WHERE published = true OR author = $auth.id
        FOR create, update WHERE author = $auth.id
        FOR delete WHERE author = $auth.id OR $auth.role = 'admin';

🧩DEFINE FIELD

Sintaks

DEFINE FIELD [ IF NOT EXISTS ] @name ON [ TABLE ] @table
    [ FLEXIBLE ]
    [ TYPE @type ]
    [ DEFAULT @value ]
    [ READONLY ]
    [ VALUE @value ]
    [ ASSERT @condition ]
    [ PERMISSIONS @permissions ]
    [ COMMENT @string ]
-- Field dasar
DEFINE FIELD name ON person TYPE string;
DEFINE FIELD age  ON person TYPE int;
DEFINE FIELD email ON person TYPE string
    ASSERT $value != NONE AND is::email($value);

-- DEFAULT
DEFINE FIELD active    ON person TYPE bool     DEFAULT true;
DEFINE FIELD created   ON person TYPE datetime DEFAULT time::now();
DEFINE FIELD score     ON player TYPE int      DEFAULT 0;

-- VALUE (computed / transform)
DEFINE FIELD full_name ON person
    VALUE $this.name.first + ' ' + $this.name.last;

-- READONLY
DEFINE FIELD id_code ON person TYPE string READONLY;

-- FLEXIBLE: field object bisa nested bebas
DEFINE FIELD metadata ON product FLEXIBLE TYPE object;

-- Optional
DEFINE FIELD bio ON person TYPE option<string>;

-- Array of typed
DEFINE FIELD tags   ON post TYPE array<string>;
DEFINE FIELD scores ON game TYPE array<int>   DEFAULT [];

-- Record link
DEFINE FIELD author ON post   TYPE record<person>;
DEFINE FIELD tags   ON post   TYPE array<record<tag>>;

-- Geometry
DEFINE FIELD location ON shop TYPE geometry<point>;
DEFINE FIELD area     ON zone TYPE geometry<polygon>;

-- Nested field
DEFINE FIELD address        ON person TYPE object;
DEFINE FIELD address.city   ON person TYPE string;
DEFINE FIELD address.postal ON person TYPE string;

-- ASSERT validasi lanjutan
DEFINE FIELD age ON person TYPE int
    ASSERT $value >= 0 AND $value <= 150;

-- Permissions per-field
DEFINE FIELD password ON user TYPE string
    PERMISSIONS
        FOR select WHERE NONE  -- tidak bisa dibaca
        FOR create, update WHERE $auth.id = id;

⚡DEFINE INDEX

DEFINE INDEX [ IF NOT EXISTS ] @name ON [ TABLE ] @table
    FIELDS | COLUMNS @fields
    [ UNIQUE ]
    [ SEARCH ANALYZER @analyzer [ BM25 [(k, b)] ] [ HIGHLIGHTS ] ]
    [ MTREE DIMENSION @d [ DIST @dist ] [ CAPACITY @c ] ]
-- Index biasa
DEFINE INDEX idx_email ON person FIELDS email;

-- UNIQUE index
DEFINE INDEX uniq_email ON user FIELDS email UNIQUE;

-- Compound index
DEFINE INDEX idx_name_age ON person FIELDS name, age;

-- Full-text search (BM25)
DEFINE ANALYZER simple_an TOKENIZERS blank FILTERS lowercase;
DEFINE INDEX idx_fts ON article FIELDS body
    SEARCH ANALYZER simple_an BM25;

-- Vector similarity (MTREE)
DEFINE INDEX idx_vec ON doc FIELDS embedding
    MTREE DIMENSION 768 DIST COSINE;

-- Hapus index
REMOVE INDEX idx_email ON person;

🎯DEFINE EVENT

DEFINE EVENT [ IF NOT EXISTS ] @name ON [ TABLE ] @table
    WHEN @condition
    THEN @query;
-- Kirim email saat user baru dibuat
DEFINE EVENT new_user ON user
    WHEN $event = "CREATE"
    THEN {
        CREATE notification CONTENT {
            to:      $value.email,
            subject: "Welcome!",
            sent:    time::now()
        };
    };

-- Log perubahan
DEFINE EVENT audit_log ON order
    WHEN $event IN ["CREATE", "UPDATE", "DELETE"]
    THEN {
        CREATE audit SET
            action  = $event,
            table   = "order",
            record  = $value.id,
            before  = $before,
            after   = $after,
            by      = $auth.id,
            at      = time::now();
    };

-- Variabel tersedia di event:
-- $event  = "CREATE" | "UPDATE" | "DELETE"
-- $value  = nilai setelah
-- $before = nilai sebelum (UPDATE/DELETE)
-- $after  = nilai sesudah (UPDATE)
-- $auth   = data sesi saat ini

⚙️DEFINE FUNCTION

DEFINE FUNCTION [ IF NOT EXISTS ] fn::@name ([ @param: @type, ... ])
    [ RETURNS @type ]
{ @body }
[ PERMISSIONS @permissions ];
-- Fungsi sederhana
DEFINE FUNCTION fn::greet($name: string) {
    RETURN "Hello, " + $name + "!";
};

-- Dengan logika bisnis
DEFINE FUNCTION fn::is_adult($person: record<person>) RETURNS bool {
    LET $age = (SELECT age FROM ONLY $person).age;
    RETURN $age >= 18;
};

-- Pemanggilan fungsi
SELECT fn::greet(name) FROM person;
RETURN fn::is_adult(person:alice);

📌DEFINE PARAM

DEFINE PARAM [ IF NOT EXISTS ] $name VALUE @value;

-- Contoh
DEFINE PARAM $app_name    VALUE "MyApp";
DEFINE PARAM $max_retries VALUE 3;

-- Penggunaan
SELECT * FROM config WHERE app = $app_name;

-- LET (sesi saja, tidak persisten)
LET $threshold = 100;
SELECT * FROM order WHERE total > $threshold;

🔐DEFINE SCOPE & TOKEN

-- DEFINE SCOPE (autentikasi custom untuk end-user)
DEFINE SCOPE account
    SESSION 24h
    SIGNUP (
        CREATE user SET
            email    = $email,
            password = crypto::argon2::generate($password)
    )
    SIGNIN (
        SELECT * FROM user
        WHERE email = $email
          AND crypto::argon2::compare(password, $password)
    );

-- DEFINE TOKEN (verifikasi JWT eksternal)
DEFINE TOKEN my_token ON NAMESPACE
    TYPE HS512
    VALUE "my_secret_key";

DEFINE TOKEN rsa_token ON DATABASE
    TYPE RS256
    VALUE "-----BEGIN PUBLIC KEY-----\n...";

👤DEFINE USER

DEFINE USER [ IF NOT EXISTS ] @name
    ON [ ROOT | NAMESPACE | DATABASE ]
    PASSWORD @password
    [ ROLES @roles ]
    [ SESSION @duration ];

-- User root baru
DEFINE USER admin ON ROOT PASSWORD 'secret123' ROLES OWNER;

-- User level database
DEFINE USER readonly ON DATABASE PASSWORD 'pass' ROLES VIEWER;

-- Roles: OWNER | EDITOR | VIEWER

🔬DEFINE ANALYZER

DEFINE ANALYZER @name
    [ TOKENIZERS @tokenizers ]
    [ FILTERS    @filters ];

-- Tokenizers: blank | camel | class | punct
-- Filters: lowercase | uppercase | ascii | ngram(min,max) | edgengram(min,max) | snowball(lang)

-- Analyzer sederhana
DEFINE ANALYZER simple
    TOKENIZERS blank
    FILTERS    lowercase;

-- Dengan snowball stemming (bahasa Inggris)
DEFINE ANALYZER english_stem
    TOKENIZERS blank
    FILTERS    lowercase, snowball(English);

-- Dengan n-gram (untuk autocomplete)
DEFINE ANALYZER autocomplete_an
    TOKENIZERS blank
    FILTERS    lowercase, edgengram(2, 10);

    RETURN "adult";
} ELSE IF $age >= 13 {
    RETURN "teen";

🛡️Permissions

-- Permissions pada tabel
DEFINE TABLE document SCHEMAFULL
    PERMISSIONS
        FOR select
            WHERE public = true
               OR owner = $auth.id
        FOR create
            WHERE $auth.id != NONE  -- user terautentikasi
        FOR update, delete
            WHERE owner = $auth.id;

-- Permissions FULL atau NONE
DEFINE TABLE public_info
    PERMISSIONS FULL;  -- siapa saja

DEFINE TABLE secret
    PERMISSIONS NONE;  -- tidak ada akses scope

-- Variabel tersedia dalam PERMISSIONS:
-- $auth    → data user yang login (dari DEFINE SCOPE)
-- $session → data sesi
-- $token   → payload token JWT
-- $scope   → nama scope
-- $value   → nilai yang sedang diproses
-- $parent  → nilai parent (untuk nested)

import Surreal from 'surrealdb.js';

const db = new Surreal();

await db.connect('ws://localhost:8000/rpc');
await db.signin({ user: 'root', pass: 'root' });
await db.use({ ns: 'myapp', db: 'production' });

// Create
const person = await db.create('person', {
    name: 'Alice',
    age: 28,
});

// Select
const people = await db.select('person');

// Query SurrealQL
const result = await db.query(
    'SELECT * FROM person WHERE age > $age',
    { age: 18 }
);

// Update
await db.merge('person:alice', { age: 29 });

// Delete
await db.delete('person:alice');

// Live query
const uuid = await db.live('person', (action, data) => {
    console.log(action, data); // CREATE / UPDATE / DELETE
});

// Stop live
await db.kill(uuid);

// Signup & signin (scope)
const token = await db.signup({
    ns: 'myapp', db: 'production', sc: 'account',
    email: 'user@example.com', password: 'secret'
});
